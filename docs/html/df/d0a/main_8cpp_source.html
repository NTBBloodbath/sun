21 
22  // NOTE: this is for debugging purposes, should be removed later
23  while (scan(state, &t)) {
24  std::string token_kind = token_str[t.type];
25  if (token_kind != "integer") {
26  sun::logger::dbg("Found token '" + token_str[t.type] + "'");
27  } else {
28  sun::logger::dbg("Found token '" + token_kind + "', value " + t.value);
29  }
30  }
31  sun::logger::dbg("Final line value " + std::to_string(state.current_ln));

33 }
